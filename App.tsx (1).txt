
import React, { useState, useEffect, useCallback } from 'react';
import { Subject, UserProgress, Question, QuizState } from './types';
import { SUBJECTS, getInitialProgress, PASS_THRESHOLD } from './constants';
import { generateQuestions } from './services/geminiService';
import Logo from './components/Logo';
import { ChevronLeft, Play, Lock, CheckCircle, Award, RefreshCcw, BookOpen, Settings } from 'lucide-react';

const App: React.FC = () => {
  const [view, setView] = useState<'home' | 'levels' | 'quiz' | 'result'>('home');
  const [selectedSubject, setSelectedSubject] = useState<Subject | null>(null);
  const [currentLevel, setCurrentLevel] = useState<number>(1);
  const [questions, setQuestions] = useState<Question[]>([]);
  const [loading, setLoading] = useState(false);
  const [progress, setProgress] = useState<UserProgress>(() => {
    const saved = localStorage.getItem('quizX_progress');
    return saved ? JSON.parse(saved) : getInitialProgress();
  });

  const [quizState, setQuizState] = useState<QuizState>({
    currentQuestionIndex: 0,
    score: 0,
    answers: Array(10).fill(null),
    isFinished: false,
    startTime: 0
  });

  const [feedback, setFeedback] = useState<{ index: number; correct: boolean } | null>(null);

  useEffect(() => {
    localStorage.setItem('quizX_progress', JSON.stringify(progress));
  }, [progress]);

  const startQuiz = async (subject: Subject, level: number) => {
    setLoading(true);
    const qs = await generateQuestions(subject, level);
    setQuestions(qs);
    setQuizState({
      currentQuestionIndex: 0,
      score: 0,
      answers: Array(qs.length).fill(null),
      isFinished: false,
      startTime: Date.now()
    });
    setLoading(false);
    setView('quiz');
  };

  const handleAnswer = (optionIndex: number) => {
    if (feedback) return;

    const currentQ = questions[quizState.currentQuestionIndex];
    const isCorrect = optionIndex === currentQ.correctAnswer;

    if (isCorrect) {
      setQuizState(prev => ({ ...prev, score: prev.score + 1 }));
    } else {
      if ('vibrate' in navigator) {
        navigator.vibrate(200);
      }
    }

    setFeedback({ index: optionIndex, correct: isCorrect });

    setTimeout(() => {
      setFeedback(null);
      if (quizState.currentQuestionIndex < questions.length - 1) {
        setQuizState(prev => ({
          ...prev,
          currentQuestionIndex: prev.currentQuestionIndex + 1,
          answers: prev.answers.map((a, i) => i === prev.currentQuestionIndex ? optionIndex : a)
        }));
      } else {
        finishQuiz();
      }
    }, 800);
  };

  const finishQuiz = () => {
    if (!selectedSubject) return;

    const finalScore = quizState.score;
    const passed = finalScore >= PASS_THRESHOLD;

    setProgress(prev => {
      const newProgress = { ...prev };
      const levels = [...newProgress[selectedSubject].levels];
      const levelIdx = currentLevel - 1;

      // Update current level stats
      levels[levelIdx] = {
        ...levels[levelIdx],
        score: Math.max(levels[levelIdx].score, finalScore),
        completed: levels[levelIdx].completed || passed
      };

      // Unlock next level if passed
      if (passed && currentLevel < 50) {
        levels[levelIdx + 1] = {
          ...levels[levelIdx + 1],
          unlocked: true
        };
      }

      newProgress[selectedSubject].levels = levels;
      return newProgress;
    });

    setView('result');
  };

  const HomeView = () => (
    <div className="min-h-screen flex flex-col items-center justify-center p-6 space-y-12 bg-black overflow-hidden">
      <div className="absolute top-0 left-0 w-full h-1/2 bg-gradient-to-b from-white/5 to-transparent pointer-events-none"></div>
      
      <div className="relative group">
        <Logo size="lg" />
        <div className="absolute -inset-4 bg-white/10 blur-3xl rounded-full group-hover:bg-white/20 transition-all duration-700"></div>
      </div>

      <div className="text-center space-y-4 relative z-10">
        <h1 className="text-5xl font-extrabold tracking-tighter text-white">QUIZ X</h1>
        <p className="text-white/50 font-medium max-w-xs mx-auto">Master GK, GS, Physics, Math & SSC with 50 levels of excellence.</p>
      </div>

      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4 w-full max-w-4xl">
        {SUBJECTS.map((sub) => (
          <button
            key={sub.id}
            onClick={() => { setSelectedSubject(sub.id); setView('levels'); }}
            className="glass p-6 rounded-3xl flex items-center space-x-4 hover:bg-white hover:text-black transition-all duration-300 transform hover:-translate-y-1 group"
          >
            <div className="p-3 bg-white/10 rounded-xl group-hover:bg-black/10 transition-colors">
              {sub.icon}
            </div>
            <div className="text-left">
              <h3 className="font-bold text-lg">{sub.name}</h3>
              <p className="text-sm opacity-60 line-clamp-1">{sub.description}</p>
            </div>
          </button>
        ))}
      </div>
      
      <div className="flex items-center space-x-6 text-white/40">
        <div className="flex flex-col items-center">
          <span className="text-xl font-bold text-white">1000+</span>
          <span className="text-xs uppercase tracking-widest">Questions</span>
        </div>
        <div className="w-px h-8 bg-white/10"></div>
        <div className="flex flex-col items-center">
          <span className="text-xl font-bold text-white">50</span>
          <span className="text-xs uppercase tracking-widest">Levels</span>
        </div>
        <div className="w-px h-8 bg-white/10"></div>
        <div className="flex flex-col items-center">
          <span className="text-xl font-bold text-white">Offline</span>
          <span className="text-xs uppercase tracking-widest">Mode</span>
        </div>
      </div>
    </div>
  );

  const LevelsView = () => {
    if (!selectedSubject) return null;
    const subjectProgress = progress[selectedSubject];

    return (
      <div className="min-h-screen p-6 bg-black text-white pb-24">
        <header className="flex items-center justify-between mb-8">
          <button onClick={() => setView('home')} className="p-2 glass rounded-full">
            <ChevronLeft />
          </button>
          <div className="text-center">
            <h2 className="text-xl font-bold uppercase tracking-widest">{selectedSubject}</h2>
            <p className="text-xs text-white/50">LEVEL SELECTION</p>
          </div>
          <div className="w-10"></div>
        </header>

        <div className="grid grid-cols-4 sm:grid-cols-5 md:grid-cols-8 gap-3 max-w-4xl mx-auto">
          {subjectProgress.levels.map((lvl) => (
            <button
              key={lvl.level}
              disabled={!lvl.unlocked}
              onClick={() => { setCurrentLevel(lvl.level); startQuiz(selectedSubject, lvl.level); }}
              className={`
                aspect-square flex flex-col items-center justify-center rounded-2xl transition-all duration-200
                ${lvl.unlocked 
                  ? lvl.completed 
                    ? 'bg-white text-black ring-4 ring-white/10' 
                    : 'glass hover:bg-white/20' 
                  : 'opacity-30 cursor-not-allowed bg-white/5'}
              `}
            >
              {!lvl.unlocked ? (
                <Lock className="w-4 h-4 mb-1" />
              ) : lvl.completed ? (
                <CheckCircle className="w-4 h-4 mb-1" />
              ) : (
                <span className="text-xs text-white/50 mb-1">LVL</span>
              )}
              <span className="text-lg font-bold">{lvl.level}</span>
            </button>
          ))}
        </div>

        <div className="fixed bottom-0 left-0 right-0 p-6 bg-gradient-to-t from-black via-black to-transparent pointer-events-none">
          <div className="max-w-4xl mx-auto glass p-4 rounded-3xl pointer-events-auto flex items-center justify-between">
            <div className="flex items-center space-x-3">
              <div className="p-2 bg-white/10 rounded-xl">
                <Award className="w-5 h-5" />
              </div>
              <div>
                <p className="text-xs text-white/50 uppercase tracking-tighter">Your Mastery</p>
                <p className="font-bold">{subjectProgress.levels.filter(l => l.completed).length} / 50 Levels</p>
              </div>
            </div>
            <div className="h-2 w-32 bg-white/10 rounded-full overflow-hidden">
              <div 
                className="h-full bg-white transition-all duration-1000" 
                style={{ width: `${(subjectProgress.levels.filter(l => l.completed).length / 50) * 100}%` }}
              ></div>
            </div>
          </div>
        </div>
      </div>
    );
  };

  const QuizView = () => {
    if (loading) {
      return (
        <div className="min-h-screen flex flex-col items-center justify-center space-y-6 bg-black">
          <Logo size="md" />
          <div className="flex flex-col items-center space-y-2">
            <div className="w-48 h-1.5 bg-white/10 rounded-full overflow-hidden">
              <div className="h-full bg-white animate-[loading_2s_infinite]"></div>
            </div>
            <p className="text-white/40 text-sm animate-pulse">Generating Challenge...</p>
          </div>
          <style>{`
            @keyframes loading {
              0% { transform: translateX(-100%); }
              100% { transform: translateX(100%); }
            }
          `}</style>
        </div>
      );
    }

    const currentQ = questions[quizState.currentQuestionIndex];
    const progressPercent = (quizState.currentQuestionIndex / questions.length) * 100;

    return (
      <div className="min-h-screen bg-black text-white p-6 flex flex-col max-w-2xl mx-auto">
        <header className="flex items-center justify-between mb-12">
          <div className="flex items-center space-x-4">
            <button onClick={() => setView('levels')} className="p-2 glass rounded-full">
              <ChevronLeft />
            </button>
            <div>
              <p className="text-xs text-white/50 uppercase font-bold tracking-widest">{selectedSubject} â€¢ LEVEL {currentLevel}</p>
              <h2 className="text-lg font-bold">Question {quizState.currentQuestionIndex + 1}/10</h2>
            </div>
          </div>
          <div className="text-right">
            <p className="text-xs text-white/50">SCORE</p>
            <p className="text-xl font-black">{quizState.score}</p>
          </div>
        </header>

        <div className="w-full h-1 bg-white/10 rounded-full mb-12">
          <div 
            className="h-full bg-white transition-all duration-500" 
            style={{ width: `${progressPercent}%` }}
          ></div>
        </div>

        <div className="flex-1 space-y-8 animate-in fade-in slide-in-from-bottom-4 duration-500">
          <div className="glass p-8 rounded-3xl min-h-[200px] flex items-center justify-center text-center">
            <h3 className="text-2xl font-bold leading-tight">{currentQ.question}</h3>
          </div>

          <div className="grid grid-cols-1 gap-4">
            {currentQ.options.map((option, idx) => {
              const isSelected = feedback?.index === idx;
              const isCorrect = idx === currentQ.correctAnswer;
              
              let bgColor = 'glass hover:bg-white/10';
              if (feedback) {
                if (isCorrect) bgColor = 'bg-green-600 border-green-400 shadow-[0_0_20px_rgba(22,163,74,0.4)]';
                else if (isSelected && !isCorrect) bgColor = 'bg-red-600 border-red-400 shadow-[0_0_20px_rgba(220,38,38,0.4)]';
              }

              return (
                <button
                  key={idx}
                  disabled={!!feedback}
                  onClick={() => handleAnswer(idx)}
                  className={`
                    p-6 rounded-2xl text-left font-semibold border border-white/5 transition-all duration-200 transform
                    ${bgColor}
                    ${!feedback && 'active:scale-95'}
                  `}
                >
                  <div className="flex items-center justify-between">
                    <span>{option}</span>
                    <div className={`w-6 h-6 rounded-full border-2 border-current flex items-center justify-center opacity-30`}>
                      <span className="text-[10px]">{String.fromCharCode(65 + idx)}</span>
                    </div>
                  </div>
                </button>
              );
            })}
          </div>
        </div>
      </div>
    );
  };

  const ResultView = () => {
    const passed = quizState.score >= PASS_THRESHOLD;
    
    return (
      <div className="min-h-screen flex flex-col items-center justify-center p-6 bg-black text-white text-center space-y-12">
        <div className="relative">
          <div className={`
            w-48 h-48 rounded-full flex flex-col items-center justify-center border-8
            ${passed ? 'border-green-500 text-green-500' : 'border-red-500 text-red-500'}
            animate-in zoom-in duration-1000
          `}>
            <span className="text-6xl font-black">{quizState.score}/10</span>
            <span className="text-xs uppercase tracking-widest font-bold">{passed ? 'Passed' : 'Try Again'}</span>
          </div>
          {passed && (
            <div className="absolute -top-4 -right-4 bg-yellow-400 text-black p-3 rounded-2xl rotate-12 animate-bounce">
              <Award className="w-8 h-8" />
            </div>
          )}
        </div>

        <div className="space-y-4">
          <h2 className="text-4xl font-black italic">{passed ? 'FANTASTIC WORK!' : 'ALMOST THERE!'}</h2>
          <p className="text-white/50 max-w-xs mx-auto">
            {passed 
              ? `You've unlocked Level ${currentLevel + 1}. Keep pushing your boundaries!`
              : `You need at least ${PASS_THRESHOLD} correct answers to unlock the next level.`}
          </p>
        </div>

        <div className="flex flex-col w-full max-w-xs space-y-4">
          {passed && currentLevel < 50 && (
            <button
              onClick={() => {
                const nextLvl = currentLevel + 1;
                setCurrentLevel(nextLvl);
                startQuiz(selectedSubject!, nextLvl);
              }}
              className="bg-white text-black p-5 rounded-3xl font-bold flex items-center justify-center space-x-2 hover:scale-105 transition-transform"
            >
              <Play className="fill-current w-5 h-5" />
              <span>NEXT LEVEL</span>
            </button>
          )}
          <button
            onClick={() => startQuiz(selectedSubject!, currentLevel)}
            className="glass p-5 rounded-3xl font-bold flex items-center justify-center space-x-2 hover:bg-white/10 transition-colors"
          >
            <RefreshCcw className="w-5 h-5" />
            <span>REPLAY LEVEL {currentLevel}</span>
          </button>
          <button
            onClick={() => setView('levels')}
            className="text-white/40 p-5 font-bold hover:text-white transition-colors"
          >
            EXIT TO LEVELS
          </button>
        </div>
      </div>
    );
  };

  return (
    <div className="selection:bg-white selection:text-black">
      {view === 'home' && <HomeView />}
      {view === 'levels' && <LevelsView />}
      {view === 'quiz' && <QuizView />}
      {view === 'result' && <ResultView />}
    </div>
  );
};

export default App;
