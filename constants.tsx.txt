export const INITIAL_LEVELS_COUNT = 50;
export const QUESTIONS_PER_LEVEL = 10;
export const PASS_THRESHOLD = 6;

export const getInitialProgress = (): UserProgress => {
  const progress: UserProgress = {};
  SUBJECTS.forEach(sub => {
    progress[sub.id] = {
      levels: Array.from({ length: INITIAL_LEVELS_COUNT }, (_, i) => ({
        level: i + 1,
        unlocked: i === 0, // Only Level 1 is unlocked initially
        score: 0,
        completed: false
      }))
    };
  });
  return progress;
};
